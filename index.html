Fotoğraf linki: ${photoUrl}` ); // Sizin WhatsApp numaranız (Örn: 905321234567) const yourNumber = '905XXXYYYYZZ'; const whatsappUrl = `https://wa.me/${yourNumber}?text=${message}`; window.open(whatsappUrl, '_blank'); } // Sayfa yüklendiğinde tüm fotoğrafları göster photoUrls.forEach((url, index) => { displayPhoto(url, index + 1); });
